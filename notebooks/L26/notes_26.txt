import matrix_helpers as M


def follow_path(board, path):
    if len(path) == 0:
        return board

    hole = M.find(board, 0)
    if hole != path[0]:
        raise ValueError(f'Bad path. Path must start at {hole}.')

    board = as_list(board)
    n = len(path)
    for i in range(1, n):
        M.swap(board, path[i-1], path[i])
    return as_tuple(board)

from functools import wraps


def save_state(f):
    @wraps(f)
    def wrapper(canvas, *args, **kwargs):
        canvas.save()
        f(canvas, *args, **kwargs)
        canvas.restore()

    return wrapper

import widget_helpers as W

from IPython.display import display


OUT = W.get_out()
CLEAR_OUTPUT = {'mouse_down': True,
                'mouse_up': True, # neu
                'mouse_move': True, # neu
                'key_down': True, # neu
                }


def try_to_call(key, args=()):
    if key in callbacks:
        f = callbacks[key]
        print(f'Calling {f.__name__}{args}')
        f(*args)


@OUT.capture(clear_output=CLEAR_OUTPUT['mouse_down'])
def on_mouse_down(x, y):
    pos = G.xy2cr(x, y, grid_spec)
    try_to_call(key='mouse_down', args=(pos,))


@OUT.capture(clear_output=CLEAR_OUTPUT['mouse_up'])  # neu
def on_mouse_up(x, y):
    pos = G.xy2cr(x, y, grid_spec)
    try_to_call(key='mouse_up', args=(pos,))


@OUT.capture(clear_output=CLEAR_OUTPUT['mouse_move'])  # neu
def on_mouse_move(x, y):
    pos = G.xy2cr(x, y, grid_spec)
    try_to_call(key='mouse_move', args=(pos,))


@OUT.capture(clear_output=CLEAR_OUTPUT['key_down'])  # neu
def on_key_down(key, *flags):
    try_to_call(key=key)


def init(canvas_, callbacks_, grid_spec_=None):
    global canvas, grid_spec, callbacks
    canvas, grid_spec, callbacks = canvas_, grid_spec_, callbacks_
    if grid_spec:
        if 'mouse_down' in callbacks:
            canvas.on_mouse_down(on_mouse_down)
        if 'mouse_up' in callbacks:  # neu
            canvas.on_mouse_up(on_mouse_up)
        if 'mouse_move' in callbacks:  # neu
            canvas.on_mouse_move(on_mouse_move)
    canvas.on_key_down(on_key_down)


def show(debug=False):
    if debug:
        display(canvas, OUT)
    else:
        display(canvas)
    canvas.focus()
