def make_matrix(ncol, nrow=None, default=None):
    '''returns a ncol x nrow matrix, or a ncol x nrow matrix if nrow=None.
       Each cell has value default.
    '''
    nrow = nrow or ncol
    matrix = [[default]*ncol for _ in range(nrow)]
    return matrix


def get_dims(matrix):
    '''returns (ncol, nrow)'''
    nrow = len(matrix)
    ncol = len(matrix[0])
    return ncol, nrow


def is_inside(pos, dims):
    x, y = pos
    ncol, nrow = dims
    return 0 <= x < ncol and 0 <= y < nrow

def get_neighbors(matrix, pos):
    '''gib eine Liste aller Positionen zurueck,
       die mit dem Feld pos eine gemeinsame Seite haben.
    '''
    x, y = pos
    dims = get_dims(matrix)
    neighbors = [(x, y-1), (x+1, y), (x, y+1), (x-1, y)]
    positions = [pos for pos in neighbors if is_inside(pos, dims)]
    return positions

def get_samecolored_4neighbors(matrix, pos0):
    '''gibt eine Liste mit allen Feldern (Positionen) zurueck,
       die den gleichen Wert wir pos0 haben, sowie eine gemeinsame Seite
    '''
    ns = M.get_neighbors(matrix, pos0, kinds='s')
    return [pos for pos in ns
            if M.get_item(matrix, pos) == M.get_item(matrix, pos0)
            ]


def get_save_neighbors(matrix, pos0):
    '''gibt [] zurueck, falls Wert des Feldes pos0 != 0 (Nachbar koennte Mine sein)
       sonst: gibt eine Liste aller Felder zurueck, die
              gemeinsame Seite oder Ecke mit dem Feld pos0 haben
    '''
    if M.get_item(matrix, pos0) != 0:
        return []
    return M.get_neighbors(matrix, pos0, kinds='sc')

def draw_matrix(canvas, matrix, grid_spec, colordict):
    '''
    colordict: Dict, der jedem Wert in der Matrix eine Farbe zuweist
    grid_spec: Grid-SPec fuer ein Gitter der gleichen Dimensionen wie die Matrix
    fuellt jedes Gitterfeld mit dem Farbwert in der Matrix
    '''
    for row, cols in enumerate(matrix):
        for col, val in enumerate(cols):
            pos = (col, row)
            color = colordict[val]
            G.fill_rect(canvas, pos, grid_spec, color=color)
