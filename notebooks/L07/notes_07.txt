def find_v3(s, char):
    '''s: String
       gibt den Index des Zeichen char in s zurueck
       oder -1 falls es nicht vorkommt
    '''
    n = len(s)
    for i in range(n):
        if s[i] == char:
            return i

    return -1

def find_v4(s, char):
    '''s: String
       gibt den Index des Zeichen char in s zurueck
       oder -1 falls es nicht vorkommt
    '''
    i = 0
    for zeichen in s:
        if zeichen == char:
            return i
        i = i + 1

    return -1
