def depth(tree):
    return max((depth(child) + 1 for child in tree[1]), default=0)

def keep_longpaths_only(tree, n):
    if n == 0:
        return tree
    children = [t for child in tree[1] if (t := keep_longpaths_only(child, n-1))]
    if children:
        return (tree[0], children)
