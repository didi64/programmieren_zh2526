def domain(method, ascii_only=True, max_len=100):
    n = 0x10ffff + 1
    if ascii_only:
        n = 256
    chars = ''
    for code_point in range(n):
        char = chr(code_point)
        if method(char):
            chars = chars + char
    print(f'{len(chars)} characters found on which {method.__name__} returns True')
    return chars[:max_len]
