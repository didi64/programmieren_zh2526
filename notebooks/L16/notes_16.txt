obstacles = [(30, 20, 10, 10), (80, 0, 10, 10)]

def check_obstacles(ball_x, ball_y):
    hits = []
    for obstacle in obstacles:
        x, y, dx, dy = obstacle
        if x <= ball_x <= x + dx and y <= ball_y <= y + dy:
            hits.append(obstacle)
    for obstacle in hits:
        obstacles.remove(obstacle)
