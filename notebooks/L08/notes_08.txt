def is_triple(wurf):
    count_dict = make_count_dict(wurf)
    values = list(count_dict.values())
    return len(wurf) == 5 and 3 in values and 1 in values

def is_two_pairs(wurf):
    count_dict = make_count_dict(wurf)
    values = list(count_dict.values())
    return len(wurf) == 5 and values.count(2) == 2
