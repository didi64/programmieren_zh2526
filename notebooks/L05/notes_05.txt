def update_heaps(heaps, move):
    '''heaps: list[int]
       move: tuple[int, int] i.e. (i, n)
       entfernt n Steine von heaps[i]
    '''
    i = move[0]
    n = move[1]
    if heaps[i] > n:
        heaps[i] = heaps[i] - n
    else:
        heaps[i] = 0
